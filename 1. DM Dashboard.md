## Stato Rapido

```dataviewjs
const gdr = await eval(await app.vault.adapter.read("z.automazioni/session_context.js"));
const count = (source, predicate) => dv.pages(source).where(p => gdr.isReal(p) && predicate(p)).length;

const cards = [
  ["Campagne", count('"Campagne"', p => p.file.name !== "Campagne" && p.stato !== "archiviata"), "Attive o da seguire"],
  ["Sessioni pronte", count('"Mondi/Sessioni"', p => p.stato === "pronto"), "Da portare al tavolo"],
  ["Missioni aperte", count('"Mondi/Missioni"', p => ["proposta", "accettata", "in corso", "pronto"].includes(p.stato)), "Scelte vive"],
  ["Clock attivi", count('"Mondi/Tracciati"', p => ["attivo", "in corso", "pronto"].includes(p.stato)), "Pressioni visibili"],
  ["Materiale pronto", count('"Mondi/Incontri" OR "Mondi/Dispense" OR "Mondi/Oggetti"', p => p.stato === "pronto"), "Usabile subito"],
  ["Idee da smistare", count('"Inbox"', p => p.file.name !== "Inbox" && !["smistata", "archiviata"].includes(p.stato)), "Appunti da decidere"]
];

const grid = dv.el("div", "", { cls: "gdr-stat-grid" });
grid.innerHTML = cards.map(([label, value, hint]) => `
  <div class="gdr-stat-card">
    <div class="gdr-stat-value">${gdr.escapeHtml(value)}</div>
    <div class="gdr-stat-label">${gdr.escapeHtml(label)}</div>
    <div class="gdr-stat-hint">${gdr.escapeHtml(hint)}</div>
  </div>
`).join("");
```

## Campagne

```dataview
TABLE stato, tono, livello_attuale, personaggi, luoghi
FROM "Campagne"
WHERE file.name != "Campagne" AND stato != "archiviata" AND stato != "conclusa" AND !startswith(file.name, "Prova -")
SORT stato ASC, nome ASC
LIMIT 8
```

## Prossime Sessioni

```dataviewjs
const gdr = await eval(await app.vault.adapter.read("z.automazioni/session_context.js"));
const linkList = value => dv.array(value ?? []).map(x => {
  const path = x?.path ?? String(x ?? "");
  const name = x?.display ?? String(x ?? "").split("/").pop();
  return `<a class="internal-link" data-href="${gdr.escapeHtml(path)}" href="${gdr.escapeHtml(path)}">${gdr.escapeHtml(name)}</a>`;
}).join(", ");

const pages = gdr.sessionCandidates(dv)
  .where(p => ["preparazione", "pronto", "in corso"].includes(p.stato))
  .sort(p => p.data ?? "9999-99-99", "asc")
  .limit(4);

if (!pages.length) {
  dv.paragraph("Nessuna sessione in preparazione.");
} else {
  const grid = dv.el("div", "", { cls: "gdr-card-grid" });
  grid.innerHTML = pages.map(p => `
    <div class="gdr-info-card">
      <div class="gdr-card-title">${gdr.internalLink(p.file)}</div>
      <div class="gdr-card-meta">${gdr.escapeHtml(p.data ?? "Senza data")} · ${gdr.escapeHtml(p.data_mondo ?? "Data mondo non indicata")} · ${gdr.escapeHtml(p.stato ?? "senza stato")}</div>
      <div class="gdr-card-line">Luoghi: ${linkList(p.luoghi) || "da scegliere"}</div>
      <div class="gdr-card-line">Personaggi: ${linkList(p.personaggi) || "da scegliere"}</div>
    </div>
  `).join("");
}
```

const gdr = await eval(await app.vault.adapter.read("z.automazioni/session_context.js"));
const maybeLink = value => {
  if (!value) return "";
  const path = value.path ?? String(value);
  const name = value.display ?? String(value).split("/").pop();

const linkList = value => dv.array(value ?? []).map(x => maybeLink(x)).join(", ");

const pages = dv.pages('"Mondi/Missioni"')

  .sort(p => p.stato ?? "zzz", "asc")
  .limit(6);

if (!pages.length) {
  dv.paragraph("Nessuna missione aperta.");
} else {
  const grid = dv.el("div", "", { cls: "gdr-card-grid" });
  grid.innerHTML = pages.map(p => `
    <div class="gdr-info-card">
      <div class="gdr-card-title">${gdr.internalLink(p.file)}</div>
      <div class="gdr-card-meta">${gdr.escapeHtml(p.stato ?? "senza stato")} · Committente: ${maybeLink(p.committente) || "non indicato"}</div>

tab: Clock

### Clock E Progress Track

```dataviewjs
const gdr = await eval(await app.vault.adapter.read("z.automazioni/session_context.js"));
const pages = dv.pages('"Mondi/Tracciati"')
  .where(p => gdr.isReal(p) && !["archiviata", "completato", "fallito"].includes(p.stato))
  .sort(p => Number(p.pressione ?? 0), "desc")
  .limit(8);

if (!pages.length) {
  dv.paragraph("Nessun clock attivo.");
} else {
  const grid = dv.el("div", "", { cls: "gdr-card-grid compact" });
  grid.innerHTML = pages.map(p => {
    const value = Math.max(0, Number(p.progress_value ?? 0));
    const max = Math.max(1, Number(p.progress_max ?? 6));
    const pct = Math.round((Math.min(value, max) / max) * 100);
    return `
      <div class="gdr-info-card compact">
        <div class="gdr-card-title">${gdr.internalLink(p.file)}</div>
        <div class="gdr-card-meta">${gdr.escapeHtml(p.tipo ?? "clock")} · ${gdr.escapeHtml(p.stato ?? "senza stato")} · pressione ${gdr.escapeHtml(p.pressione ?? 0)}</div>
        <div class="gdr-track-bar"><span style="width: ${pct}%"></span></div>
        <div class="gdr-card-line">${value}/${max} · ${gdr.escapeHtml(p.innesco ?? "innesco non indicato")}</div>
        <div class="gdr-card-line">${gdr.escapeHtml(p.prossima_mossa ?? "prossima mossa non indicata")}</div>
      </div>
    `;
  }).join("");
}
```
