## Stato Rapido

```dataviewjs
const count = (source, predicate) => dv.pages(source).where(predicate).length;
const escapeHtml = value => String(value ?? "").replace(/[&<>"']/g, c => ({ "&": "&amp;", "<": "&lt;", ">": "&gt;", "\"": "&quot;", "'": "&#39;" }[c]));

const cards = [
  ["Sessioni pronte", count('"Mondo/Sessioni"', p => p.stato === "pronto"), "Da portare al tavolo"],
  ["PNG in gioco", count('"Mondo/Personaggi"', p => p.tipo === "png" && p.stato === "in gioco"), "Presenze attive"],
  ["Missioni aperte", count('"Mondo/Missioni"', p => ["proposta", "accettata", "in corso"].includes(p.stato)), "Scelte vive"],
  ["Incontri pronti", count('"Mondo/Incontri"', p => ["pronto", "in gioco"].includes(p.stato)), "Scene preparate"],
  ["Idee da smistare", count('"Inbox"', p => !["smistata", "archiviata"].includes(p.stato) && p.file.name !== "Inbox"), "Inbox"],
  ["Bozze", count('"Mondo" OR "Campagne"', p => p.stato === "bozza"), "Da completare"]
];

const grid = dv.el("div", "", { cls: "gdr-stat-grid" });
grid.innerHTML = cards.map(([label, value, hint]) => `
  <div class="gdr-stat-card">
    <div class="gdr-stat-value">${escapeHtml(value)}</div>
    <div class="gdr-stat-label">${escapeHtml(label)}</div>
    <div class="gdr-stat-hint">${escapeHtml(hint)}</div>
  </div>
`).join("");
```

## Carte Operative

### Prossime Sessioni

```dataviewjs
const escapeHtml = value => String(value ?? "").replace(/[&<>"']/g, c => ({ "&": "&amp;", "<": "&lt;", ">": "&gt;", "\"": "&quot;", "'": "&#39;" }[c]));
const internalLink = file => `<a class="internal-link" data-href="${escapeHtml(file.path)}" href="${escapeHtml(file.path)}">${escapeHtml(file.name)}</a>`;
const linkList = value => dv.array(value ?? []).map(x => {
  const path = x?.path ?? String(x ?? "");
  const name = x?.display ?? String(x ?? "").split("/").pop();
  return `<a class="internal-link" data-href="${escapeHtml(path)}" href="${escapeHtml(path)}">${escapeHtml(name)}</a>`;
}).join(", ");

const pages = dv.pages('"Mondo/Sessioni"')
  .where(p => ["preparazione", "pronto"].includes(p.stato))
  .sort(p => p.data ?? "9999-99-99", "asc")
  .limit(4);

if (!pages.length) {
  dv.paragraph("Nessuna sessione in preparazione.");
} else {
  const grid = dv.el("div", "", { cls: "gdr-card-grid" });
  grid.innerHTML = pages.map(p => `
    <div class="gdr-info-card">
      <div class="gdr-card-title">${internalLink(p.file)}</div>
      <div class="gdr-card-meta">${escapeHtml(p.data ?? "Senza data")} · ${escapeHtml(p.data_mondo ?? "Data mondo non indicata")} · ${escapeHtml(p.stato ?? "senza stato")}</div>
      <div class="gdr-card-line">Luoghi: ${linkList(p.luoghi) || "da scegliere"}</div>
      <div class="gdr-card-line">Personaggi: ${linkList(p.personaggi) || "da scegliere"}</div>
    </div>
  `).join("");
}
```

### Missioni Aperte

```dataviewjs
const escapeHtml = value => String(value ?? "").replace(/[&<>"']/g, c => ({ "&": "&amp;", "<": "&lt;", ">": "&gt;", "\"": "&quot;", "'": "&#39;" }[c]));
const internalLink = file => `<a class="internal-link" data-href="${escapeHtml(file.path)}" href="${escapeHtml(file.path)}">${escapeHtml(file.name)}</a>`;
const maybeLink = value => {
  if (!value) return "";
  const path = value.path ?? String(value);
  const name = value.display ?? String(value).split("/").pop();
  return `<a class="internal-link" data-href="${escapeHtml(path)}" href="${escapeHtml(path)}">${escapeHtml(name)}</a>`;
};
const linkList = value => dv.array(value ?? []).map(x => maybeLink(x)).join(", ");

const pages = dv.pages('"Mondo/Missioni"')
  .where(p => ["proposta", "accettata", "in corso"].includes(p.stato))
  .sort(p => p.stato ?? "zzz", "asc")
  .limit(6);

if (!pages.length) {
  dv.paragraph("Nessuna missione aperta.");
} else {
  const grid = dv.el("div", "", { cls: "gdr-card-grid" });
  grid.innerHTML = pages.map(p => `
    <div class="gdr-info-card">
      <div class="gdr-card-title">${internalLink(p.file)}</div>
      <div class="gdr-card-meta">${escapeHtml(p.stato ?? "senza stato")} · Committente: ${maybeLink(p.committente) || "non indicato"}</div>
      <div class="gdr-card-line">Luoghi: ${linkList(p.luoghi) || "nessuno"}</div>
      <div class="gdr-card-line">Ricompense: ${linkList(p.ricompense) || "non indicate"}</div>
    </div>
  `).join("");
}
```

### Materiale Pronto

```dataviewjs
const escapeHtml = value => String(value ?? "").replace(/[&<>"']/g, c => ({ "&": "&amp;", "<": "&lt;", ">": "&gt;", "\"": "&quot;", "'": "&#39;" }[c]));
const internalLink = file => `<a class="internal-link" data-href="${escapeHtml(file.path)}" href="${escapeHtml(file.path)}">${escapeHtml(file.name)}</a>`;
const maybeLink = value => {
  if (!value) return "";
  const path = value.path ?? String(value);
  const name = value.display ?? String(value).split("/").pop();
  return `<a class="internal-link" data-href="${escapeHtml(path)}" href="${escapeHtml(path)}">${escapeHtml(name)}</a>`;
};

const pages = dv.pages('"Mondo/Incontri" OR "Mondo/Dispense" OR "Mondo/Oggetti"')
  .where(p => p.stato === "pronto")
  .sort(p => p.categoria ?? "zzz", "asc")
  .limit(8);

if (!pages.length) {
  dv.paragraph("Nessun materiale segnato come pronto.");
} else {
  const grid = dv.el("div", "", { cls: "gdr-card-grid compact" });
  grid.innerHTML = pages.map(p => `
    <div class="gdr-info-card compact">
      <div class="gdr-card-title">${internalLink(p.file)}</div>
      <div class="gdr-card-meta">${escapeHtml(p.categoria ?? "nota")} · ${escapeHtml(p.tipo ?? "tipo non indicato")}</div>
      <div class="gdr-card-line">Luogo: ${maybeLink(p.luogo) || "non indicato"}</div>
    </div>
  `).join("");
}
```
